---
title: "Get started with Uni ECharts"
description: "Install Uni ECharts via npm or the uni-modules plugin market and render your first Apache ECharts chart in a uni-app project in a few steps."
---

Uni ECharts supports two installation methods: **npm** (recommended) and **uni-modules** via the DCloud plugin market. Both methods let you render Apache ECharts charts across Web, WeChat/Alipay Mini Programs, and APP. Choose the method that fits your project setup.

<Info>
**Prerequisites**
- Node.js >= 20.19.0, >= 22.12.0, or >= 24.0.0
- echarts >= 5.3.0
- vue >= 3.3.0 (uni-app has not yet adapted to Vue 3.5; use `3.4.x` to stay aligned with uni-app)
</Info>

<Tabs>
  <Tab title="NPM (recommended)">

    <Steps>
      <Step title="Install packages">
        Install both `echarts` and `uni-echarts` from npm.

        <CodeGroup>
          ```bash pnpm
          pnpm add echarts uni-echarts
          ```
          ```bash yarn
          yarn add echarts uni-echarts
          ```
          ```bash npm
          npm install echarts uni-echarts
          ```
        </CodeGroup>

        <Tip>
          If you need a custom ECharts build, see [Configuring ECharts](/guide/echarts#custom-echarts-build).
        </Tip>
      </Step>

      <Step title="Configure Vite">
        Because the `uni-echarts` npm package ships as uncompiled `.vue` files, Vite's dependency pre-bundling can produce extra `echarts` copies. Exclude `uni-echarts` from pre-bundling:

        ```js vite.config.js
        import { defineConfig } from "vite";

        export default defineConfig({
          // ...
          optimizeDeps: { // [!code ++]
            exclude: [ // [!code ++]
              "uni-echarts" // [!code ++]
            ] // [!code ++]
          } // [!code ++]
        });
        ```

        <Note>
          Since `2.5.0`, adding the Vite plugin (see below) automatically adds this exclusion — you can skip the manual `optimizeDeps` config.
        </Note>

        **Vite plugin (recommended since 2.0.0)**

        The Uni ECharts Vite plugin handles `optimizeDeps` exclusion, auto-calls `provideEcharts`, and enables custom ECharts builds.

        ```js vite.config.js
        import { UniEcharts } from "uni-echarts/vite"; // [!code ++]
        import { defineConfig } from "vite";

        export default defineConfig({
          // ...
          plugins: [
            UniEcharts() // [!code ++]
          ]
        });
        ```
      </Step>

      <Step title="Set up auto-import (optional)">
        Uni ECharts works with [`@uni-helper/vite-plugin-uni-components`](https://github.com/uni-helper/vite-plugin-uni-components) and [`unplugin-auto-import`](https://github.com/unplugin/unplugin-auto-import) for zero-import usage.

        <CodeGroup>
          ```bash pnpm
          pnpm add -D @uni-helper/vite-plugin-uni-components unplugin-auto-import
          ```
          ```bash yarn
          yarn add --dev @uni-helper/vite-plugin-uni-components unplugin-auto-import
          ```
          ```bash npm
          npm install -D @uni-helper/vite-plugin-uni-components unplugin-auto-import
          ```
        </CodeGroup>

        ```js vite.config.js
        import Uni from "@dcloudio/vite-plugin-uni";
        import UniComponents from "@uni-helper/vite-plugin-uni-components";
        import { UniEchartsResolver } from "uni-echarts/resolver"; // [!code ++]
        import AutoImport from "unplugin-auto-import/vite";
        import { defineConfig } from "vite";

        export default defineConfig({
          plugins: [
            AutoImport({
              resolvers: [
                UniEchartsResolver() // [!code ++]
              ]
            }),
            // Ensure this is placed before Uni()
            UniComponents({
              resolvers: [
                UniEchartsResolver() // [!code ++]
              ]
            }),
            Uni()
          ]
        });
        ```

        If you use PNPM, add this to your `.npmrc` (see [issue #389](https://github.com/unplugin/unplugin-vue-components/issues/389)):

        ```dotenv .npmrc
        shamefully-hoist=true
        ```

        For TypeScript type hints for auto-imported components, add to `tsconfig.json`:

        ```json5 tsconfig.json
        {
          "compilerOptions": {
            "types": [
              // ...
              "uni-echarts/global" // [!code ++]
            ]
          }
        }
        ```
      </Step>

      <Step title="Use the component">
        Call `provideEcharts` with your echarts instance, register the modules you need, then pass `option` to `<uni-echarts>`.

        <Note>
          `provideEcharts` is required. If you configured the Vite plugin in step 2, this call is handled automatically and you can omit it.
        </Note>

        ```vue
        <template>
          <uni-echarts custom-class="chart" :option="option"></uni-echarts>
        </template>

        <script setup>
        import { PieChart } from "echarts/charts";
        import { DatasetComponent, LegendComponent, TooltipComponent } from "echarts/components";
        import * as echarts from "echarts/core";
        import { CanvasRenderer } from "echarts/renderers";
        import { provideEcharts } from "uni-echarts/shared";
        import { ref } from "vue";

        // Required: provide the echarts instance
        provideEcharts(echarts);

        echarts.use([
          LegendComponent,
          TooltipComponent,
          DatasetComponent,
          PieChart,
          CanvasRenderer
        ]);

        const option = ref({
          legend: { top: 10, left: "center" },
          tooltip: { trigger: "item" },
          series: [{
            type: "pie",
            radius: ["30%", "52%"],
            itemStyle: { borderWidth: 2, borderColor: "#ffffff", borderRadius: 10 }
          }],
          dataset: {
            dimensions: ["Source", "Count"],
            source: [
              ["Search Engine", 1048],
              ["Direct", 735],
              ["Email", 580]
            ]
          }
        });
        </script>

        <style>
        .chart {
          height: 300px;
        }
        </style>
        ```

        <Accordion title="Options API example">
          ```vue
          <template>
            <uni-echarts custom-class="chart" :option="option"></uni-echarts>
          </template>

          <script>
          import { PieChart } from "echarts/charts";
          import { DatasetComponent, LegendComponent, TooltipComponent } from "echarts/components";
          import * as echarts from "echarts/core";
          import { CanvasRenderer } from "echarts/renderers";
          import UniEcharts from "uni-echarts";
          import { ECHARTS_KEY } from "uni-echarts/shared";
          import { defineComponent } from "vue";

          echarts.use([LegendComponent, TooltipComponent, DatasetComponent, PieChart, CanvasRenderer]);

          export default defineComponent({
            components: { UniEcharts },
            provide: {
              [ECHARTS_KEY]: echarts
            },
            data() {
              return {
                option: {
                  legend: { top: 10, left: "center" },
                  tooltip: { trigger: "item" },
                  series: [{ type: "pie", radius: ["30%", "52%"] }],
                  dataset: {
                    dimensions: ["Source", "Count"],
                    source: [["Search Engine", 1048], ["Direct", 735]]
                  }
                }
              };
            }
          });
          </script>

          <style>
          .chart { height: 300px; }
          </style>
          ```
        </Accordion>

        <Warning>
          If your chart is not displaying on Mini Programs, check the [Mini Program class/style FAQ](/guide/faq#mini-program-class--style-attributes-are-not-working) for solutions.
        </Warning>
      </Step>
    </Steps>

  <Tab title="Uni Modules">

    <Steps>
      <Step title="Install ECharts via npm">
        <CodeGroup>
          ```bash pnpm
          pnpm add echarts
          ```
          ```bash yarn
          yarn add echarts
          ```
          ```bash npm
          npm install echarts
          ```
        </CodeGroup>

        <Tip>
          If you need a custom ECharts build, see [Configuring ECharts](/guide/echarts#custom-echarts-build).
        </Tip>
      </Step>

      <Step title="Download from the plugin market">
        Download the [Uni ECharts plugin](https://ext.dcloud.net.cn/plugin?id=22035) from the DCloud uni-app plugin market and import it into your project.

        <Tip>
          Uni ECharts supports [easycom](https://uniapp.dcloud.net.cn/component/#easycom). When using the uni-modules method, the `<uni-echarts>` component is available without any import statements.
        </Tip>

        <Warning>
          The component's API functions still need to be imported manually. Note that the import path differs from the npm method.
        </Warning>
      </Step>

      <Step title="Use the component">
        Import `provideEcharts` from the uni-modules path (not from `uni-echarts/shared`).

        <Note>
          Since `2.0.0`, `provideEcharts` must be called manually even when using uni-modules. The component no longer auto-fills a default echarts instance.
        </Note>

        ```vue
        <template>
          <uni-echarts custom-class="chart" :option="option"></uni-echarts>
        </template>

        <script setup>
        import { PieChart } from "echarts/charts";
        import { DatasetComponent, LegendComponent, TooltipComponent } from "echarts/components";
        import * as echarts from "echarts/core";
        import { CanvasRenderer } from "echarts/renderers";
        // Note: import path differs from npm method
        import { provideEcharts } from "@/uni_modules/xiaohe-echarts";
        import { ref } from "vue";

        // Required since 2.0.0
        provideEcharts(echarts);

        echarts.use([
          LegendComponent,
          TooltipComponent,
          DatasetComponent,
          PieChart,
          CanvasRenderer
        ]);

        const option = ref({
          legend: { top: 10, left: "center" },
          tooltip: { trigger: "item" },
          series: [{ type: "pie", radius: ["30%", "52%"] }],
          dataset: {
            dimensions: ["Source", "Count"],
            source: [["Search Engine", 1048], ["Direct", 735]]
          }
        });
        </script>

        <style>
        .chart {
          height: 300px;
        }
        </style>
        ```
